163 template <typename StringType, typename... ParameterTypes>
164 template <std::size_t N>
165 FLY_CONSTEVAL BasicFormatString<StringType, ParameterTypes...>::BasicFormatString(
166  const char_type (&format)[N]) noexcept :
167  m_lexer(format)
168 {
169  std::optional<char_type> ch;
170 
171  if constexpr (!(BasicFormatTraits::is_formattable_v<ParameterTypes> && ...))
172  {
173  on_error("An overloaded operator<< must be defined for all format parameters");
174  }
175 
176  while (!has_error() && (ch = m_lexer.consume()))
177  {
178  if (ch == s_left_brace)
179  {
180  if (m_lexer.consume_if(s_left_brace))
181  {
182  continue;
183  }
184  else if (m_specifier_count >= m_specifiers.size())
185  {
186  on_error("Exceeded maximum allowed number of specifiers");
187  }
188  else if (auto specifier = parse_specifier(SpecifierType::Full); specifier)
189  {
190  m_specifiers[m_specifier_count++] = *std::move(specifier);
191  }
192  }
193  else if (ch == s_right_brace)
194  {
195  if (m_lexer.consume_if(s_right_brace))
196  {
197  continue;
198  }
199 
200  on_error("Closing brace } must be esacped");
201  }
202  }
203 }

240 template <typename StringType, typename... ParameterTypes>
241 constexpr auto
242 BasicFormatString<StringType, ParameterTypes...>::parse_specifier(SpecifierType specifier_type)
243  -> std::optional<FormatSpecifier>
244 {
245  // The opening { will have already been consumed, so the starting position is one less.
246  const auto starting_position = m_lexer.position() - 1;
247 
248  FormatSpecifier specifier {};
249  specifier.m_position = parse_position();
250 
251  auto param_type = parameter_type(specifier.m_position);
252 
253  if (param_type == ParameterType::Generic)
254  {
255  // TODO: For now, generic format parameters must be formatted with "{}".
256  if (!m_lexer.consume_if(s_right_brace))
257  {
258  on_error("Generic types must be formatted with {}");
259  return std::nullopt;
260  }
261  }
262  else
263  {
264  if (param_type)
265  {
266  if ((specifier_type == SpecifierType::Full) && m_lexer.consume_if(s_colon))
267  {
268  specifier.parse(*this, *param_type);
269 
270  if (has_error())
271  {
272  return std::nullopt;
273  }
274  }
275  else
276  {
277  specifier.template infer_type<BasicFormatString>(*param_type);
278  }
279  }
280 
281  if (!m_lexer.consume_if(s_right_brace))
282  {
283  on_error("Detected unclosed format string - must end with }");
284  return std::nullopt;
285  }
286  }
287 
288  if (m_expect_no_positions_specified && m_expect_all_positions_specified)
289  {
290  on_error("Argument position must be provided on all or not on any specifier");
291  }
292  else if (!param_type)
293  {
294  on_error("Argument position exceeds number of provided arguments");
295  }
296 
297  specifier.m_size = m_lexer.position() - starting_position;
298  return specifier;
299 }

318 template <typename StringType, typename... ParameterTypes>
319 template <size_t N>
320 constexpr auto BasicFormatString<StringType, ParameterTypes...>::parameter_type(size_t index)
321  -> std::optional<ParameterType>
322 {
323  if constexpr (N < sizeof...(ParameterTypes))
324  {
325  if (N != index)
326  {
327  return parameter_type<N + 1>(index);
328  }
329 
330  using T = std::decay_t<std::tuple_element_t<N, std::tuple<ParameterTypes...>>>;
331 
332  if constexpr (is_supported_character_v<T>)
333  {
334  return ParameterType::Character;
335  }
336  else if constexpr (is_like_supported_string_v<T>)
337  {
338  return ParameterType::String;
339  }
340  else if constexpr (std::is_pointer_v<T> || std::is_null_pointer_v<T>)
341  {
342  return ParameterType::Pointer;
343  }
344  else if constexpr (
345  BasicFormatTraits::is_integer_v<T> || BasicFormatTraits::is_default_formatted_enum_v<T>)
346  {
347  return ParameterType::Integral;
348  }
349  else if constexpr (std::is_floating_point_v<T>)
350  {
351  return ParameterType::FloatingPoint;
352  }
353  else if constexpr (std::is_same_v<T, bool>)
354  {
355  return ParameterType::Boolean;
356  }
357  else
358  {
359  return ParameterType::Generic;
360  }
361  }
362  else
363  {
364  return std::nullopt;
365  }
366 }
